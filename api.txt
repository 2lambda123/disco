
/* DiscoDB C API */

/* Construct DiscoDB */

ddb_cons_t *ddb_new();
void ddb_add(ddb_cons_t *c, const ddb_entry_t *key, const ddb_entry_t *values, uint32_t num_values);
const char *ddb_finalize(ddb_cons_t *c, uint64_t *length);

/* Use DiscoDB */

const ddb_t *ddb_loads(const char *data, uint64_t length);
const char *ddb_dumps(const ddb_t *db, uint64_t *length);
ddb_cursor_t *ddb_keys(const ddb_t *db);
ddb_cursor_t *ddb_values(const ddb_t *db);
ddb_cursor_t *ddb_getitem(const ddb_t *db, const ddb_entry_t *key);
ddb_cursor_t *ddb_query(const ddb_t *ddb, const ddb_term_t *terms, uint32_t length);
uint32_t ddb_resultset_size(const ddb_cursor_t *cur); /* works only with ddb_getitem, ddb_keys and ddb_values cursors */
ddb_entry_t *ddb_next(ddb_cursor_t *cur);

typedef struct{
        const uint8_t *data;
        uint32_t length;
} ddb_entry_t;

typedef struct{
        ddb_key_t *keys;
        uint32_t length;
} ddb_term_t;

typedef struct{
        ddb_entry_t key;
        uint not;
} ddb_key_t;
